# Clone the ultralytics repository
git clone https://github.com/ultralytics/ultralytics

# Navigate to the cloned directory
cd ultralytics

# Install the package in editable mode for development
pip install -e .

pip install ultralytics --no-deps

# Install other core dependencies
pip install numpy matplotlib pandas pyyaml pillow psutil requests tqdm scipy seaborn ultralytics-thop

# Install headless OpenCV instead of the default

pip install -U openvino
pip install opencv-python-headless
pip install py-cpuinfo
pip install einops


pip3 install torch torchvision torchaudio --index-url https://download.pytorch.org/whl/cu128

# 安装核心依赖
pip install onnx==1.17.0 onnxruntime==1.18.0 onnx-simplifier==0.4.35
# 可选：安装GPU版ONNX Runtime（如果有NVIDIA GPU）
pip install onnxruntime-gpu==1.18.0

pip uninstall ultralytics 
pip install -e .
